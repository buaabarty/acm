bool readword(string &a)
{
    a = ""; char c;
    if (scanf("%c", &c) == EOF) return false;
    while (!isalpha(c))
        if (scanf("%c", &c) == EOF) return false;
    while (isalpha(c))
    {
        a = a + (char)tolower(c);
        if (scanf("%c", &c) == EOF) return false;
    }
    return true;
}

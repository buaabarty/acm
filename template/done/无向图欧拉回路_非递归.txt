int stack[M], top, len;
void dfs(int v)
{
    stack[++top] = v;
    while (top)
    {
        int x = stack[top];
        bool flag = true;
        for (int i = p[x]; i != -1; i = e[i].next)
            if (e[i].f)

                e[i].f = false;
